"use client"

import { useEffect, useRef } from "react"
import Image from "next/image"
import { Button } from "@/components/ui/button"
import { ArrowRight, Shield, Clock, Gem, Headphones } from "lucide-react"
import gsap from "gsap"

export function HeroSection() {
  const sectionRef = useRef<HTMLElement>(null)
  const robotRef = useRef<HTMLDivElement>(null)
  const contentRef = useRef<HTMLDivElement>(null)
  const moonRef = useRef<HTMLDivElement>(null)
  const trustBarRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    const ctx = gsap.context(() => {
      gsap.from(robotRef.current, {
        y: 100,
        opacity: 0,
        duration: 1.2,
        delay: 0.3,
        ease: "power3.out",
      })

      gsap.from(contentRef.current, {
        y: 50,
        opacity: 0,
        duration: 1,
        delay: 0.5,
        ease: "power3.out",
      })

      gsap.from(moonRef.current, {
        scale: 0.5,
        opacity: 0,
        duration: 1.5,
        delay: 0.2,
        ease: "power4.out",
      })

      gsap.from(trustBarRef.current, {
        y: 30,
        opacity: 0,
        duration: 1,
        delay: 1,
        ease: "power3.out",
      })
    }, sectionRef)

    return () => ctx.revert()
  }, [])

  return (
    <section 
      ref={sectionRef} 
      className="relative min-h-screen bg-[#050508] overflow-hidden"
    >
      {/* Background Grid */}
      <div 
        className="absolute inset-0 opacity-30"
        style={{
          backgroundImage: `
            linear-gradient(rgba(255, 0, 127, 0.08) 1px, transparent 1px),
            linear-gradient(90deg, rgba(255, 0, 127, 0.08) 1px, transparent 1px)
          `,
          backgroundSize: "80px 80px",
        }}
      />

      {/* Ambient Glow */}
      <div className="absolute top-1/4 left-1/4 w-[600px] h-[600px] bg-[#ff007f]/5 rounded-full blur-[150px] pointer-events-none" />

      {/* ===== MAIN CONTAINER - MAX WIDTH LOCKED ===== */}
      <div className="relative w-full max-w-[1400px] mx-auto min-h-screen">
        
        {/* Left Content - Absolutely positioned within container */}
        <div
          ref={contentRef}
          className="absolute left-4 sm:left-8 md:left-12 lg:left-16 top-1/2 -translate-y-1/2 z-30 w-[90%] sm:w-[70%] md:w-[55%] lg:w-[45%] pt-16"
        >
          <h2 className="text-4xl sm:text-5xl md:text-6xl lg:text-7xl font-black text-white uppercase tracking-tighter leading-[0.9] mb-4 sm:mb-6">
            STOP RENTING SOFTWARE.
            <br />
            <span className="text-transparent bg-clip-text bg-linear-to-r from-[#ff007f] to-[#ff6b6b]">
              START OWNING AI ASSETS.
            </span>
          </h2>

          <p className="text-gray-400 text-sm sm:text-base md:text-lg mb-6 sm:mb-8 leading-relaxed max-w-md">
            We build proprietary AI infrastructure that belongs to{" "}
            <span className="text-white font-medium">you</span>. 
            Stop leasing your company's intelligence to big tech and secure your long-term valuation.
          </p>

          <Button className="bg-[#ff007f] hover:bg-[#ff007f]/90 text-white rounded-full px-6 sm:px-8 md:px-10 py-4 sm:py-5 md:py-6 text-sm sm:text-base font-bold shadow-[0_0_40px_rgba(255,0,127,0.4)] hover:shadow-[0_0_60px_rgba(255,0,127,0.6)] transition-all duration-300">
            BOOK STRATEGY CALL <ArrowRight className="w-5 h-5 ml-2" />
          </Button>

          {/* Trust Bar */}
          <div ref={trustBarRef} className="mt-8 sm:mt-10">
            <div className="flex flex-wrap gap-4 sm:gap-6">
              <TrustItem icon={Shield} value="100%" label="IP Ownership" />
              <TrustItem icon={Clock} value="4-8 Wks" label="Launch Time" />
              <TrustItem icon={Gem} value="0%" label="Equity Taken" />
              <div className="hidden sm:block">
                <TrustItem icon={Headphones} value="24/7" label="Support" />
              </div>
            </div>
          </div>
        </div>

        {/* ===== MOON + ROBOT - Fixed to bottom right of container ===== */}
        <div className="absolute bottom-0 right-0 w-[70%] sm:w-[60%] md:w-[55%] lg:w-[50%] h-full flex items-end justify-center">
          
          {/* Moon - Centered behind robot, z-index 1 */}
          <div
            ref={moonRef}
            className="absolute z-[1]"
            style={{ 
              width: "clamp(400px, 80vw, 900px)",
              height: "clamp(400px, 80vw, 900px)",
              bottom: "-50%",
              left: "50%",
              transform: "translateX(-50%)",
            }}
          >
            {/* Outer ambient glow */}
            <div 
              className="absolute rounded-full"
              style={{
                inset: "-30%",
                background: "radial-gradient(circle, rgba(255,0,127,0.25) 0%, rgba(255,0,127,0.1) 40%, transparent 70%)",
                filter: "blur(60px)",
              }}
            />
            {/* Moon ring - pink glowing border */}
            <div 
              className="absolute inset-0 rounded-full"
              style={{
                border: "2px solid rgba(255,0,127,0.5)",
                boxShadow: "0 0 100px 20px rgba(255,0,127,0.3), inset 0 0 80px rgba(255,0,127,0.15)",
              }}
            />
            {/* Moon body - dark circle */}
            <div className="absolute inset-0 rounded-full bg-[#030305]" />
            {/* Inner gradient for depth */}
            <div 
              className="absolute rounded-full"
              style={{
                inset: "4px",
                background: "linear-gradient(to top, #050508 0%, #0a0a12 50%, #0f0f1a 100%)",
              }}
            />
            {/* Top edge pink highlight */}
            <div 
              className="absolute inset-0 rounded-full"
              style={{
                background: "linear-gradient(to bottom, rgba(255,0,127,0.25) 0%, rgba(255,0,127,0.08) 30%, transparent 60%)",
              }}
            />
          </div>

          {/* Robot - z-index 5, above moon */}
          <div
            ref={robotRef}
            className="relative z-[5] w-[90%] sm:w-[85%] md:w-[80%] max-w-[600px]"
          >
            <Image
              src="/images/robot.png"
              alt="AI Robot"
              width={750}
              height={750}
              priority
              className="w-full h-auto object-contain"
              style={{ filter: "drop-shadow(0 0 60px rgba(255, 0, 127, 0.5))" }}
            />
          </div>
        </div>
      </div>

      {/* ===== BOTTOM FADE - Full width, z-index 10 ===== */}
      <div 
        className="absolute bottom-0 left-0 right-0 h-32 sm:h-40 z-[10] pointer-events-none"
        style={{
          background: "linear-gradient(to top, #050508 0%, #050508 30%, transparent 100%)"
        }}
      />
    </section>
  )
}

// Trust Item Component for cleaner code
function TrustItem({ icon: Icon, value, label }: { icon: React.ElementType; value: string; label: string }) {
  return (
    <div className="flex items-center gap-2 sm:gap-3">
      <div className="w-8 h-8 sm:w-10 sm:h-10 rounded-lg bg-[#ff007f]/10 border border-[#ff007f]/20 flex items-center justify-center">
        <Icon className="w-4 h-4 sm:w-5 sm:h-5 text-[#ff007f]" />
      </div>
      <div>
        <p className="text-white font-bold text-sm sm:text-base">{value}</p>
        <p className="text-gray-500 text-[10px] sm:text-xs">{label}</p>
      </div>
    </div>
  )
}
